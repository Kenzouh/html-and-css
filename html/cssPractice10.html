div5 = display:inline-block; (Aligns itself, but still a block element)
span1 = display:inline; (doesn't do anything because span is already an inline element, but it aligns itself to the divs thanks to "inline" value)
span2 = display:block; (becomes a block element, takes whole width space)
span3 = display:inline-block;
span4 = display:none;
span6 = display: inline-block; visibility: hidden;
span7 = display: inline;
